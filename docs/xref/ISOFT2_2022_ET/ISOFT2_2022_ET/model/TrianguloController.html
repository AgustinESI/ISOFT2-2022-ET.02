6   public class TrianguloController {
7   
8   	public TrianguloController() {
9   		super();
10  	}
11  
12  	public String tipoDeTriangulo(Triangulo t) {
13  		String result = "";
14  		// Ifs para comprobar que los valroes no sean negativos y 0
15  		if (t.getLado1() <= 0 || t.getLado2() <= 0 || t.getLado3() <= 0 || t.getAngulo1() <= 0 || t.getAngulo2() <= 0 || t.getAngulo3() <= 0) {
16  			throw new NumerosValidosException("Numeros negativos no validos");
17  		}
18  
19  		// Ifs para comprobar que cumpla la condicion de triangulo
20  		if ((t.getAngulo1() + t.getAngulo2() + t.getAngulo3()) > 180) {
21  			throw new AngulosValidosException("Angulos no validos");
22  		}
23  
24  		// Ifs para comprobar que tipo es segun los lados
25  		if (t.getLado1() == t.getLado2() && t.getLado1() == t.getLado3()) {
26  			result += "Equilatero ";
27  		} else if (t.getLado1() != t.getLado2() && t.getLado1() != t.getLado3() && t.getLado2() != t.getLado3()) {
28  			result += "Escaleno ";
29  		} else {
30  			result += "Isosceles ";
31  		}
32  		// Ifs para comprobar que tipo es segun los angulos
33  		if (t.getAngulo1() == 90 || t.getAngulo2() == 90 || t.getAngulo3() == 90) {
34  			result += "Rectangulo";
35  		} else if (t.getAngulo1() < 90 && t.getAngulo2() < 90 && t.getAngulo3() < 90) {
36  			result += "Acutangulo";
37  		} else {
38  			result += "Obstusangulo";
39  		}
40  		return result;
41  	}
42  
43  }
